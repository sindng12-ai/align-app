'flex' : 'none'; if(actionActive) actionActive.style.display = 'flex'; if(actionHidden) actionHidden.style.display = 'none'; } if(typeof m_renderPartnerRadar === 'function') m_renderPartnerRadar(); if(typeof mo_renderLevelUpQueue === 'function') mo_renderLevelUpQueue(); };
